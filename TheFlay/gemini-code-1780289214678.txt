# --- PANTHEON SCRIPT REGISTRY ---

# Helper function to invoke scripts
function Invoke-PantheonScript {
    param([string]$ScriptName, [Parameter(ValueFromRemainingArguments=$true)]$Args)
    $ScriptPath = "$env:USERPROFILE\Scripts\$ScriptName.ps1"
    if (Test-Path $ScriptPath) { & $ScriptPath @Args } else { Write-Error "Script $ScriptName not found." }
}
